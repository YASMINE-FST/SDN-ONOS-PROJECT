Dans Karaf —changer settings :

# 1. flowTimeout = 0 (règles permanentes)
cfg set org.onosproject.fwd.ReactiveForwarding flowTimeout 0

# 2. matchIpv4Address = true (règles basées sur IP)
cfg set org.onosproject.fwd.ReactiveForwarding matchIpv4Address true

# 3. allowExtraneousRules = true (garder les règles manuelles)
cfg set org.onosproject.net.flow.impl.FlowRuleManager allowExtraneousRules true

# 4. portStatsPollFrequency = 2 (stats toutes les 2 sec)
cfg set org.onosproject.provider.of.device.impl.OpenFlowDeviceProvider portStatsPollFrequency 2

cfg get org.onosproject.fwd.ReactiveForwarding

# Voir tous les settings modifiés
